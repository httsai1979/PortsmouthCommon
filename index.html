# [PERFORMANCE STRATEGY] Netlify Server Configuration [build] command = "npm run build" publish = "dist" # 1. 啟用 Asset 優化 (自動壓縮圖片與文字檔案) [build.processing] skip_processing = false [build.processing.css] bundle = true minify = true [build.processing.js] bundle = true minify = true [build.processing.html] pretty_urls = true # 2. 設定快取策略 (Caching Headers) # 對於 index.html，我們不快取，確保使用者永遠看到最新版本 [[headers]] for = "/*" [headers.values] Cache-Control = "public, max-age=0, must-revalidate" # 對於 assets (JS/CSS/Images)，我們設定「長效快取」 (Immutable) # 因為 Vite 會產生 hash 檔名 (e.g., index-XyZ.js)，內容變了檔名就會變，所以可以放心永久快取 [[headers]] for = "/assets/*" [headers.values] Cache-Control = "public, max-age=31536000, immutable"
